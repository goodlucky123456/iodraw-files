```mermaid
import networkx as nx
import matplotlib.pyplot as plt

# 创建图
G = nx.Graph()

# 添加节点（区分带△和不带△的节点）
core_node = "中心节点"
triangle_nodes = [f"△{i}" for i in range(5)]  # 带△的节点
other_nodes = [f"普通节点{i}" for i in range(3)]  # 不带△的节点
all_nodes = [core_node] + triangle_nodes + other_nodes
G.add_nodes_from(all_nodes)

# 添加边（黑色实线）
edges_solid = [
    (core_node, triangle_nodes[0]),
    (core_node, triangle_nodes[1]),
    (core_node, triangle_nodes[2]),
    (core_node, triangle_nodes[3]),
    (core_node, triangle_nodes[4]),
    (triangle_nodes[1], other_nodes[0]),
    (triangle_nodes[3], other_nodes[1]),
    (triangle_nodes[4], other_nodes[2]),
]
G.add_edges_from(edges_solid)

# 添加边（蓝色虚线）
edges_dashed = [
    (core_node, triangle_nodes[1]),
    (core_node, triangle_nodes[3]),
    (core_node, triangle_nodes[4]),
]

# 绘制节点
pos = nx.spring_layout(G, seed=42)  # 固定布局（seed保证每次一致）

# 绘制黑色实线边
nx.draw_networkx_edges(G, pos, edgelist=edges_solid, edge_color="black", width=2)
# 绘制蓝色虚线边
nx.draw_networkx_edges(G, pos, edgelist=edges_dashed, edge_color="blue", style="dashed", width=2)

# 绘制节点（中心节点+带△节点+普通节点）
nx.draw_networkx_nodes(G, pos, nodelist=[core_node], node_size=500, node_color="white", edgecolors="black")
nx.draw_networkx_nodes(G, pos, nodelist=triangle_nodes, node_size=500, node_color="white", edgecolors="black")
nx.draw_networkx_nodes(G, pos, nodelist=other_nodes, node_size=500, node_color="white", edgecolors="black")

# 给带△的节点添加三角形标记
for node in triangle_nodes:
    x, y = pos[node]
    plt.scatter(x, y, marker="$\Delta$", color="red", s=200)  # 绘制红色△

# 绘制红色圈（手动调整坐标范围）
ax = plt.gca()
circle = plt.Circle((0.5, 0.5), 0.7, color="red", linestyle="solid", linewidth=2, fill=False)
ax.add_artist(circle)

# 隐藏坐标轴
plt.axis("off")
plt.show()
```
